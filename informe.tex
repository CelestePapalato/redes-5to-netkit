\section{SWITCH}


\textit{Topología}:

PC1[0]=”A”

PC2[0]=”B”

SWITCH[0]=”A”

SWITCH[1]=”B”

\textbf{En PC1}:

ifconfig eth0 192.168.0.1/24 up


\textbf{En PC2}:

ifconfig eth0 192.168.0.2/24 up

\textbf{En SWITCH}:

ifconfig eth0 up

ifconfig eth1 up

brctl add br0

brctl addif br0 eth1

brctl addif br0 eth0

ifconfig br0 up
